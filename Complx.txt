import sys
import matplotlib
matplotlib.use('Agg')

import matplotlib.pyplot as plt
import numpy as np

def iter_num(c, max_n=100):
    n = 0
    z = 0
    while n < max_n:
        z = z**2 + c
        n += 1
        if abs(z) > 2:
            break
    return n

pituus = 500

koordx = []
koordy = []

for x in range(-pituus,pituus):
    for y in range(-pituus,pituus):
        c = x/100 + 1j * y/100
        n = iter_num(c)
        if n == 100:
            koordx.append(x/100)
            koordy.append(y/100)


x = np.array(koordx)
y = np.array(koordy)

plt.scatter(x, y)
plt.show()

plt.scatter(x, y)
plt.show()
